    ''' play sound effecfs
        Note that you need to run with "sudo"

        Pidog.speak(name, volume=100)

    import os
    import time

    # change working directory

    # my_dog.speak("angry")

        my_dog.speak(name)

        time.sleep(3) # Note that the duration of each sound effect is different
